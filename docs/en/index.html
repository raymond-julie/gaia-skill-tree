In a hurry?

shell 
# Install
pip install gaia-registry

# Initialise your profile inside a Git repo
gaia init --user <your-github-handle>

# Scan the repo for skill evidence
gaia scan

# Promote a candidate from the scan output
gaia promote <skill-id>

# View your tree
gaia tree
